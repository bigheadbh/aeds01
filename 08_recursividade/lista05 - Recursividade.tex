\begin{center}
{\Large \bf Lista de Exercícios 05}
\end{center}

\begin{enumerate}
	\item Fazer um método recursivo que recebe um número inteiro e retorna o seu fatorial.
	
	
	\item Fazer um método recursivo que recebe um número inteiro n e retorna o n-ésimo termo da equação de recorrência abaixo:

	$T(1) = 2$

	$T(2) = 3$

	$T(n) = 5 * n + T(n - 1)^{n}$


	\item Fazer um método recursivo que recebe um número inteiro e positivo n e calcula o somatório abaixo.

	$n + (n-1) + ... + 1 + 0$

	\item Fazer um método recursivo que imprima de um número natural em base binária
	
	\item Fazer um método recursivo que multiplique dois números naturais, através de somas sucessivas
	
	\item Fazer um método recursivo que calcule o MDC (máximo divisor comum) de dois inteiros positivos $m$ e $n$
	
	\item Fazer um método recursivo que conte os dígitos de um determinado número.
	
	\item Fazer um método recursivo que determine se um número é ou não primo.
	
	
\end{enumerate}
